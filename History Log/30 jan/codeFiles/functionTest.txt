// This procedure swaps
// values of X and Y
FUNCTION SWAP(X : INTEGER, Y : INTEGER)
    DECLARE Temp : INTEGER
    Temp ← X // temporarily store X
    X ← Y
    Y ← Temp
    RETURN 0
ENDFUNCTION

DECLARE X : INTEGER
DECLARE N : INTEGER
DECLARE P : INTEGER
X ← 9
N ← 5

P ← SWAP(X,N)
